import React, { useState } from 'react';

export default function App() {
  const [formData, setFormData] = useState({
    campaignName: '',
    targetAudience: '',
    budget: '',
    platform: 'linkedin',
    adCopy: ''
  });
  const [showSuccess, setShowSuccess] = useState(false);

  const handleChange = (e) => {
    const { id, value } = e.target;
    setFormData(prev => ({
      ...prev,
      [id]: value
    }));
  };

  const handleBudgetChange = (e) => {
    const value = e.target.value.replace(/[^0-9]/g, '');
    setFormData(prev => ({
      ...prev,
      budget: value
    }));
  };

  const handleSubmit = (e) => {
    e.preventDefault();
    console.log('Campaign Data:', formData);
    setShowSuccess(true);
    setTimeout(() => setShowSuccess(false), 3000);
  };

  return (
    <div style={{ fontFamily: "'Plus Jakarta Sans', sans-serif", minHeight: '100vh', display: 'flex', flexDirection: 'column', background: 'linear-gradient(135deg, #eff6ff 0%, #ffffff 50%, #dbeafe 100%)' }}>
      {/* Header */}
      <header style={{ background: 'rgba(255, 255, 255, 0.8)', backdropFilter: 'blur(10px)', borderBottom: '1px solid #e5e7eb', position: 'sticky', top: 0, zIndex: 50 }}>
        <div style={{ maxWidth: '1280px', margin: '0 auto', padding: '1rem 1.5rem', display: 'flex', alignItems: 'center', justifyContent: 'space-between' }}>
          <div style={{ display: 'flex', alignItems: 'center', gap: '0.75rem' }}>
            <div style={{ width: '40px', height: '40px', background: 'linear-gradient(135deg, #2563eb 0%, #1d4ed8 100%)', borderRadius: '0.5rem', display: 'flex', alignItems: 'center', justifyContent: 'center', color: 'white', fontSize: '1.5rem' }}>
              ⚡
            </div>
            <span style={{ fontSize: '1.25rem', fontWeight: 'bold', color: '#111827' }}>Campaign Predictor Lite</span>
          </div>
          <nav style={{ display: 'flex', gap: '2rem' }}>
            <a href="#about" style={{ color: '#374151', textDecoration: 'none', fontWeight: 500 }}>About</a>
            <a href="#how-it-works" style={{ color: '#374151', textDecoration: 'none', fontWeight: 500 }}>How It Works</a>
          </nav>
        </div>
      </header>

      {/* Main Content */}
      <main style={{ flex: 1, display: 'flex', alignItems: 'center', justifyContent: 'center', padding: '3rem 1rem' }}>
        <div style={{ width: '100%', maxWidth: '48rem' }}>
          {/* Header Text */}
          <div style={{ textAlign: 'center', marginBottom: '2rem' }}>
            <div style={{ fontSize: '0.875rem', fontWeight: 600, color: '#2563eb', marginBottom: '0.75rem' }}>Step 1/2</div>
            <h1 style={{ fontSize: '3rem', fontWeight: 'bold', color: '#111827', marginBottom: '0.75rem', lineHeight: 1.2 }}>Campaign Predictor Lite</h1>
            <p style={{ fontSize: '1.125rem', color: '#374151', marginBottom: '0.5rem' }}>Analyze Your Marketing Campaign Before You Launch.</p>
            <p style={{ color: '#4b5563', fontSize: '1rem' }}>Enter your details below.</p>
          </div>

          {/* Form Card */}
          <div style={{ background: 'white', borderRadius: '1rem', boxShadow: '0 10px 40px rgba(0, 0, 0, 0.1)', padding: '2rem' }}>
            <form onSubmit={handleSubmit}>
              {/* Campaign Name */}
              <div style={{ marginBottom: '1.5rem' }}>
                <label style={{ display: 'flex', alignItems: 'center', gap: '0.5rem', fontSize: '0.875rem', fontWeight: 600, color: '#111827', marginBottom: '0.75rem' }}>
                  📄 Campaign Name
                </label>
                <input 
                  type="text" 
                  id="campaignName"
                  value={formData.campaignName}
                  onChange={handleChange}
                  placeholder="e.g., AI Chatbot Launch"
                  required
                  style={{ width: '100%', padding: '0.75rem 1rem', border: '1px solid #d1d5db', borderRadius: '0.5rem', fontFamily: "'Plus Jakarta Sans', sans-serif", fontSize: '1rem', transition: 'all 0.3s ease' }}
                />
              </div>

              {/* Target Audience */}
              <div style={{ marginBottom: '1.5rem' }}>
                <label style={{ display: 'flex', alignItems: 'center', gap: '0.5rem', fontSize: '0.875rem', fontWeight: 600, color: '#111827', marginBottom: '0.75rem' }}>
                  👥 Target Audience
                </label>
                <input 
                  type="text" 
                  id="targetAudience"
                  value={formData.targetAudience}
                  onChange={handleChange}
                  placeholder="e.g., Tech startups, 18-40"
                  required
                  style={{ width: '100%', padding: '0.75rem 1rem', border: '1px solid #d1d5db', borderRadius: '0.5rem', fontFamily: "'Plus Jakarta Sans', sans-serif", fontSize: '1rem' }}
                />
              </div>

              {/* Budget and Platform Row */}
              <div style={{ display: 'grid', gridTemplateColumns: '1fr 1fr', gap: '1rem', marginBottom: '1.5rem' }}>
                {/* Budget */}
                <div>
                  <label style={{ display: 'flex', alignItems: 'center', gap: '0.5rem', fontSize: '0.875rem', fontWeight: 600, color: '#111827', marginBottom: '0.75rem' }}>
                    💵 Budget ($)
                  </label>
                  <div style={{ display: 'flex', border: '1px solid #d1d5db', borderRadius: '0.5rem', overflow: 'hidden' }}>
                    <div style={{ padding: '0.75rem 1rem', background: '#f3f4f6', color: '#374151', fontWeight: 500, display: 'flex', alignItems: 'center' }}>$</div>
                    <input 
                      type="text" 
                      id="budget"
                      value={formData.budget}
                      onChange={handleBudgetChange}
                      placeholder="e.g., 5000"
                      required
                      style={{ flex: 1, border: 'none', padding: '0.75rem 1rem', fontFamily: "'Plus Jakarta Sans', sans-serif", fontSize: '1rem' }}
                    />
                  </div>
                </div>

                {/* Platform */}
                <div>
                  <label style={{ display: 'flex', alignItems: 'center', gap: '0.5rem', fontSize: '0.875rem', fontWeight: 600, color: '#111827', marginBottom: '0.75rem' }}>
                    🔗 Platform
                  </label>
                  <select 
                    id="platform"
                    value={formData.platform}
                    onChange={handleChange}
                    required
                    style={{ width: '100%', padding: '0.75rem 1rem', border: '1px solid #d1d5db', borderRadius: '0.5rem', fontFamily: "'Plus Jakarta Sans', sans-serif", fontSize: '1rem' }}
                  >
                    <option value="linkedin">LinkedIn</option>
                    <option value="facebook">Facebook</option>
                    <option value="twitter">Twitter</option>
                    <option value="instagram">Instagram</option>
                    <option value="google">Google Ads</option>
                  </select>
                </div>
              </div>

              {/* Ad Copy */}
              <div style={{ marginBottom: '1.5rem' }}>
                <label style={{ display: 'flex', alignItems: 'center', gap: '0.5rem', fontSize: '0.875rem', fontWeight: 600, color: '#111827', marginBottom: '0.75rem' }}>
                  📊 Ad Copy
                </label>
                <textarea 
                  id="adCopy"
                  value={formData.adCopy}
                  onChange={handleChange}
                  placeholder="Type or paste your ad copy here... (Introducing ChatBot Pro...)"
                  required
                  style={{ width: '100%', padding: '0.75rem 1rem', border: '1px solid #d1d5db', borderRadius: '0.5rem', fontFamily: "'Plus Jakarta Sans', sans-serif", fontSize: '1rem', minHeight: '150px', resize: 'none' }}
                />
              </div>

              {/* Analyze Button */}
              <button 
                type="submit"
                style={{ width: '100%', background: 'linear-gradient(135deg, #2563eb 0%, #1d4ed8 100%)', color: 'white', border: 'none', padding: '0.75rem 1.5rem', borderRadius: '0.5rem', fontSize: '1rem', fontWeight: 'bold', cursor: 'pointer', display: 'flex', alignItems: 'center', justifyContent: 'center', gap: '0.5rem', marginTop: '1.5rem', transition: 'all 0.3s ease' }}
              >
                ⚡ ANALYZE CAMPAIGN
              </button>
            </form>
          </div>
        </div>
      </main>

      {/* Footer */}
      <footer style={{ background: 'rgba(255, 255, 255, 0.5)', borderTop: '1px solid #e5e7eb', padding: '1.5rem', textAlign: 'center' }}>
        <p style={{ fontSize: '0.875rem', color: '#4b5563' }}>Copyright © 2023. All rights reserved.</p>
      </footer>

      {/* Success Message */}
      {showSuccess && (
        <div style={{ position: 'fixed', top: '20px', right: '20px', background: '#10b981', color: 'white', padding: '1rem 1.5rem', borderRadius: '0.5rem', boxShadow: '0 10px 25px rgba(0, 0, 0, 0.2)', zIndex: 1000, animation: 'slideIn 0.3s ease' }}>
          ✓ Campaign analyzed successfully!
        </div>
      )}
    </div>
  );
}
